		String answer = "";
		s += " ";

		StringBuilder tmp = new StringBuilder();
		for(int i = 0; i < s.length(); i++) {
			if(i != 0 && i % 7 == 0) {
				int num = Integer.parseInt(tmp.toString(), 2);
				answer += (char) num;
				tmp.setLength(0);
			}
			if (s.charAt(i) == '#') {
				tmp.append(1);
			} else if (s.charAt(i) == '*') {
				tmp.append(0);
			}

		}
		return answer;
	}
